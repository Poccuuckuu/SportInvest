    <script>
        // Улучшенный код для мобильного меню
        document.addEventListener('DOMContentLoaded', function() {
            const mobileMenuBtn = document.getElementById('mobileMenuBtn');
            const mobileMenu = document.getElementById('mobileMenu');
            
            // Функция для переключения меню
            function toggleMenu() {
                mobileMenu.classList.toggle('active');
                
                // Блокировка прокрутки тела страницы при открытом меню
                if (mobileMenu.classList.contains('active')) {
                    document.body.style.overflow = 'hidden';
                } else {
                    document.body.style.overflow = '';
                }
            }
            
            // Обработчик клика по кнопке меню
            mobileMenuBtn.addEventListener('click', function(e) {
                e.stopPropagation();
                toggleMenu();
            });
            
            // Закрытие меню при клике на пункт или вне меню
            document.addEventListener('click', function(e) {
                if (mobileMenu.classList.contains('active') && 
                    !mobileMenu.contains(e.target) && 
                    e.target !== mobileMenuBtn) {
                    toggleMenu();
                }
            });
            
            // Обработчики для пунктов меню
            const menuLinks = document.querySelectorAll('.mobile-nav-link');
            menuLinks.forEach(link => {
                link.addEventListener('click', function() {
                    toggleMenu();
                });
            });
            
            // Код для калькулятора и FAQ остается без изменений
            const rangeInput = document.getElementById('investmentRange');
            const currentValue = document.getElementById('currentValue');
            const profitAmount = document.getElementById('profitAmount');
            
            if (rangeInput && currentValue && profitAmount) {
                rangeInput.addEventListener('input', () => {
                    const value = parseInt(rangeInput.value);
                    const formattedValue = new Intl.NumberFormat('ru-RU').format(value);
                    currentValue.textContent = formattedValue + '₽';
                    
                    const profit = Math.round(value * 0.02);
                    const formattedProfit = new Intl.NumberFormat('ru-RU').format(profit);
                    profitAmount.textContent = formattedProfit + '₽';
                });
            }
            
            const faqQuestions = document.querySelectorAll('.faq-question');
            if (faqQuestions) {
                faqQuestions.forEach(question => {
                    question.addEventListener('click', () => {
                        const item = question.parentNode;
                        item.classList.toggle('active');
                        
                        faqQuestions.forEach(q => {
                            if (q !== question) {
                                q.parentNode.classList.remove('active');
                            }
                        });
                    });
                });
            }
        });
    </script>
